173 void testBasicMath() {
174 Awkccc_variable two("2"), eleven("11"), ans;
175 // Addition
176 ans=two+eleven;
177 CPPUNIT_ASSERT(fabs(ans.number_)-13 < Awkccc_variable::epsilon_);
178 ans=two+11.0;
179 CPPUNIT_ASSERT(fabs(ans.number_)-13 < Awkccc_variable::epsilon_);
180 ans=2+eleven;
181 CPPUNIT_ASSERT(fabs(ans.number_)-13 < Awkccc_variable::epsilon_);
182 // Subtraction
183 ans=eleven-two;
184 CPPUNIT_ASSERT(fabs(ans.number_)-9 < Awkccc_variable::epsilon_);
185 ans=eleven-2.0;
186 CPPUNIT_ASSERT(fabs(ans.number_)-9 < Awkccc_variable::epsilon_);
187 ans=11-two;
188 CPPUNIT_ASSERT(fabs(ans.number_)-9 < Awkccc_variable::epsilon_);
189 // Multiply
190 ans=eleven*two;
191 CPPUNIT_ASSERT(fabs(ans.number_)-22 < Awkccc_variable::epsilon_);
192 ans=eleven*2.0;
193 CPPUNIT_ASSERT(fabs(ans.number_)-22 < Awkccc_variable::epsilon_);
194 ans=11*two;
195 CPPUNIT_ASSERT(fabs(ans.number_)-22 < Awkccc_variable::epsilon_);
196 // Division
197 ans=eleven/two;
198 CPPUNIT_ASSERT(fabs(ans.number_)-5.5 < Awkccc_variable::epsilon_);
199 ans=eleven/2.0;
200 CPPUNIT_ASSERT(fabs(ans.number_)-5.5 < Awkccc_variable::epsilon_);
201 ans=11/two;
202 CPPUNIT_ASSERT(fabs(ans.number_)-5.5 < Awkccc_variable::epsilon_);
203 }

205 void testIncDec() {
206 Awkccc_variable countup("13");
207 CPPUNIT_ASSERT(countup ++ - 13 < Awkccc_variable::epsilon_);
208 CPPUNIT_ASSERT(countup - 14 < Awkccc_variable::epsilon_);
209 CPPUNIT_ASSERT(++ countup - 15 < Awkccc_variable::epsilon_);
210 CPPUNIT_ASSERT(countup - 15 < Awkccc_variable::epsilon_);
211 Awkccc_variable countdown(100);
212 CPPUNIT_ASSERT(countdown -- - 100 < Awkccc_variable::epsilon_);
213 CPPUNIT_ASSERT(countdown - 99 < Awkccc_variable::epsilon_);
214 CPPUNIT_ASSERT(-- countdown - 98 < Awkccc_variable::epsilon_);
215 CPPUNIT_ASSERT(countdown - 98 < Awkccc_variable::epsilon_);
216 }
