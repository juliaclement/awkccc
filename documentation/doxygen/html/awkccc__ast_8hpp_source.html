178 , extra_children_allowed_(true)

184 }
185 ast_node( node_types type, jclib::CountedPointer<Symbol> sym, int rule_nr = -1, bool dummy = false)
186 : type_( type )
187 , name_(sym->name_)
188 , sym_(sym)
189 , has_sym_( sym.isset() )

192 , rule_nr_( rule_nr )
193 , sibling_nodes_()
194 , child_nodes_()

196 }
197 ast_node * add_sibling(jclib::CountedPointer<ast_node> adoptee) {
198 if( rule_nr_ == -1 )
199 rule_nr_ = adoptee->rule_nr_;
200 if( ! has_sym_ && adoptee->has_sym_ ) {
201 sym_ = adoptee->sym_;
202 has_sym_ = adoptee->has_sym_;
203 }
204 sibling_nodes_.push_back(adoptee);
205 return this;
206 }
207 ast_node * add_child(jclib::CountedPointer<ast_node> adoptee) {
208 if( rule_nr_ == -1 )
209 rule_nr_ = adoptee->rule_nr_;
210 if( ! has_sym_ && adoptee->has_sym_ ) {
211 sym_ = adoptee->sym_;
212 has_sym_ = adoptee->has_sym_;
213 }
214 child_nodes_.push_back(adoptee);
215 return this;
216 }
217 ast_node * add_children( std::initializer_list< jclib::CountedPointer<ast_node> > adoptees);
218
219 /*** Recursively promote siblings to parent (if allowed) */
220 virtual void clean_tree(ast_node * parent);
