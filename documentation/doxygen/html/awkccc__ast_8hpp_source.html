185 class ast_node: public jclib::CountedPointerTarget {
186 public:
187 node_types type_;
188 jclib::jString name_;
189 int rule_nr_;
190 bool has_sym_;
191 bool extra_children_allowed_;
192 bool dummy_;
193 jclib::CountedPointer<Symbol> sym_;
194 std::vector<jclib::CountedPointer<ast_node> > sibling_nodes_;
195 std::vector<jclib::CountedPointer<ast_node> > child_nodes_;
196 // Despatcher for Visitors
197 virtual void accept( ast_node_visitor * visitor ){
198 visitor->visit_ast_node( this );}
199 ast_node( node_types type, jclib::jString name , int rule_nr = -1, bool dummy = false)
200 : type_( type )
201 , name_( name )
202 , rule_nr_( rule_nr )
203 , sym_( nullptr )
204 , has_sym_( false )
205 , extra_children_allowed_(true)
206 , dummy_( dummy )
207 , sibling_nodes_()
208 , child_nodes_()
209 {
210 sym_ = SymbolTable::instance().get(name_, 0, VARIABLE );
211 }
212 ast_node( char * full_name, node_types type, Symbol *sym, int rule_nr = -1, bool dummy = false)
213 : type_( type )
214 , name_(sym->awk_name_)
215 , sym_(sym)
216 , has_sym_( sym != nullptr )
217 , extra_children_allowed_(true)
218 , dummy_( dummy )
219 , sibling_nodes_()
220 , child_nodes_()
221 , rule_nr_( rule_nr )
222 {
223 }
224 ast_node( node_types type, jclib::CountedPointer<Symbol> sym, int rule_nr = -1, bool dummy = false)
225 : type_( type )
226 , name_(sym->awk_name_)
227 , sym_(sym)
228 , has_sym_( sym.isset() )
229 , extra_children_allowed_(true)
230 , dummy_( dummy )
231 , rule_nr_( rule_nr )
232 , sibling_nodes_()
233 , child_nodes_()
234 {
235 }
236 ast_node * add_sibling(jclib::CountedPointer<ast_node> adoptee) {
237 if( rule_nr_ == -1 )
238 rule_nr_ = adoptee->rule_nr_;
239 if( ! has_sym_ && adoptee->has_sym_ ) {
240 sym_ = adoptee->sym_;
241 has_sym_ = adoptee->has_sym_;
242 }
243 sibling_nodes_.push_back(adoptee);
244 return this;
245 }
246 ast_node * add_child(jclib::CountedPointer<ast_node> adoptee) {
247 if( rule_nr_ == -1 )
248 rule_nr_ = adoptee->rule_nr_;
249 if( ! has_sym_ && adoptee->has_sym_ ) {
250 sym_ = adoptee->sym_;
251 has_sym_ = adoptee->has_sym_;
252 }
253 child_nodes_.push_back(adoptee);
254 return this;
255 }
256 ast_node * add_children( std::initializer_list< jclib::CountedPointer<ast_node> > adoptees);
257
258 /*** Recursively promote siblings to parent (if allowed) */
259 virtual void clean_tree(ast_node * parent);
260 };
