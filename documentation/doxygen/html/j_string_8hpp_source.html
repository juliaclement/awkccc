121 #else
122 // **** This MUST be the last data member in the struct as
123 // we deliberately allocate padding then overrun the declared
124 // length of the array.
125 //
126 // I think it's safe as:
127 // C99 requires that struct members are laid out in
128 // declaration order:
129 // "Within a structure object, the non-bit-field members
130 // and the units in which bit-fields reside have addresses
131 // that increase in the order in which they are declared"
132 // C++ probably requires that in a standard-layout struct
133 // object struct members are laid out in declaration order.
134 // Problem is I can't seem to locate it in an actual C++ standard
135 // but every compiler I know works that way.
136 //
137 // If you get strange crashes, try defining SAFE_JSTRING above
138 char data_[1];
139 #endif

269 jString pad(std::size_t length, char pad_char = ' ' ) const {
270 if( length <= len())
271 return *this;
272
273 jStringBase * new_stringBase = jStringBase::create(get()->data_, length );
274

276 for( ; pad_ptr < new_stringBase->data_ + length; ++ pad_ptr ) {
277 *pad_ptr = pad_char;

279 *pad_ptr = '\0';

283 /*** Split string on delim, returning answers as an array of strings,
284 * once max_splits copies of the delimiter have been found, the
285 * remainder if any is a single string at the end of the array.
286 */
287 std::vector<jString> split(const char * delim, int max_splits=std::numeric_limits<int>::max() ) const {
288 std::vector<jString> answer;
289 const char * buffer = get()->data_;
290 const char * buffer_end = buffer+len();
291 const int step = strlen(delim);
292 auto position = strstr( buffer, delim );
293 if( position == nullptr ){
294 answer.push_back(*this);
295 return answer;
296 }
297 for( ;
298 position != nullptr && max_splits != 0;
299 --max_splits ){
300 jString chunk(buffer,position);
301 answer.push_back( chunk );
302 buffer += chunk.len() + step;
303 position = (buffer < buffer_end) ? strstr( buffer, delim ) : nullptr;
304 }
305 if( buffer < buffer_end ) {
306 answer.push_back(jString(buffer));
307 }
308 return answer;
309 }

359 template<class T> jString join( const T & input ) {
360 size_t total_length = 0;
361 std::vector<jString> strings; // No guarantee that input can be parsed twice
362 for( jString bit : input ) {
363 strings.push_back( bit );
364 total_length += bit.len() + len();
365 }
366 total_length = total_length + 1 - len();
367 jStringBase * answer = jStringBase::create(total_length);
368 char * out = const_cast<char *>(answer->data_);
369 bool first = true;
370 for( jString bit : strings ) {
371 if( first )
372 first = false;
373 else {
374 memmove(out, data(), len());
375 out+=len();
376 }
377 memmove(out, bit.data(), bit.len());
378 out+=bit.len();
379 }
380 *out='\0';
381 return answer;
382 }
