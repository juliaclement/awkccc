20 enum addressing {
21 // High order 3 bits of op code [0..7]
22 STACK_RELATIVE = 0,
23 VARIABLE_STORE,
24 LOCAL_STORE, // Current function local variables
25 ARRAY_INDEX, // TOS = Array, 2ndOS = Index in Array
26 FUNCTIONS,
27 BUILT_IN,
28 IMMEDIATE, // low order 23 bits of op code as unsigned. Range = 0..8,388,607
29 GUARD3 = IMMEDIATE
30 };
31 enum opcodes {
32 // First 32 (Through GUARD5) shared as lexer tokens
33 // First 64 (Through GUARD6) shared as low order 6 bits of opcode in interpreter
34 LOAD_NUM=0, // move operand to top of stack as number
35 LOAD_STR, // move operand to top of stack as string
36 LOAD_REGEX, // move operand to top of stack as regex
37 LOADNEGATIVE, // move negated numeric operand to top of stack
38 STORE, // = (Assign operator for results of computation)
39 POP, // remove N elements from stack. 0 = discard one, >=1 = convert to std::Vector and leave pointer on top of stack
40 ADD,
41 SUB,
42 MUL,
43 DIV,
44 MODULO,
45 INCR, // ++
46 DECR, // --
47 XOR,
48 OR,
49 AND,
50 CONCAT, // string concatenate
51 // Compare TOS against value, set TRUE/FALSE flag.
52 LE,
53 LT,
54 GT,
55 GE,
56 EQ,
57 NE, // = NOT EQ
58 MATCH, // regex match (sets TRUE/FALSE flag)
59 NOTMATCH, // regex match fails (sets TRUE/FALSE flag)
60 // The following 3 ops don't use memory other than TOS & TRUE/FALSE flag
61 RETURN, // Return a value
62 NEGATE,
63 NOT,
64 // End of dual purpose tokens
65 GUARD5 = NOT,
66 CALL,
67 CALL_CODE, // Call code procedure (e.g. cos, substr, etc)
68 BRANCH,
69 BRANCH_TRUE,
70 BRANCH_FALSE,
71 // The following ops don't use memory other than TOS & TRUE/FALSE flag
72 ENTER, // Enter a procedure
73 DUP, // Duplicate TOS
74 PUSH_TEST_RESULT, // True -> 1, False -> 0
75 SWAP, // exchange TOS & 2nd on stack
76 GUARD6 // Tested to make sure that all interpreter ops fit in 6 bits
77 };
78 enum tokens {
79 // Non-interpreter
80 LBRACE='{',
81 RBRACE='}',
82 LBRACKET='[',
83 RBRACKET=']',
84 LPAREN='(',

86 };
87 static_assert(int(addressing::GUARD3) < 8);
