772**
773** + The state number for the parser at this level of the stack.

1415 YYCODETYPE iLookAhead, /* The look-ahead token */
1416 YYACTIONTYPE stateno /* Current state number */
1417){

1419
1420 if( stateno>YY_MAX_SHIFT ) return stateno;

1424#endif

1427 assert( i>=0 );
1428 assert( i<=YY_ACTTAB_COUNT );
1429 assert( i+YYNTOKEN<=(int)YY_NLOOKAHEAD );
1430 assert( iLookAhead!=YYNOCODE );
1431 assert( iLookAhead < YYNTOKEN );
