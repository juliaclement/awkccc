35class SymbolTableImpl : public SymbolTable {
36 public:
37 map_t sym_table_;
38 virtual void insert(jclib::CountedPointer<Symbol>s) {
39 sym_table_[s->name_] = s;
40 }
41
42 virtual Symbol * insert(
43 class jclib::jString name,
44 int token,
45 SymbolType type = VARIABLE) {
46 Symbol * sym = new Symbol( name, token, type);
47 sym_table_[sym->name_] = sym;
48 return sym;
49 }
50
51 virtual Symbol * find( jclib::jString &str) {
52 auto search = sym_table_.find(str);
53 if (search == sym_table_.end()) {
54 return nullptr;
55 }
56 return search->second.get();
57 }
58
62 virtual Symbol * get(
63 class jclib::jString name,
64 int token,
65 SymbolType type = VARIABLE) {
66 if( Symbol * answer = find(name) )
67 return answer;
68 return insert( name, token, type );
69 }
70
71 static SymbolTableImpl & instance() {
72 static SymbolTableImpl the_table;
73 return the_table;
74 }
75
76 virtual void load(std::initializer_list<struct _Symbol_loader> input) {
77 for( auto i : input) {
78 insert( i.name_, i.token_, i.type_);
79 }
80 }
81
82 virtual void loadnamespace(const jclib::jString &namespace_name,
83 bool also_load_global,
84 std::initializer_list<struct _Symbol_loader> input) {
85 size_t max_name_len = 0;
86 for( auto i : input) {
87 max_name_len = std::max<size_t>(max_name_len,strlen(i.name_));
88 }
89 auto tmp_name = new char[namespace_name.len()+max_name_len+3];
90 strcpy(tmp_name,(const char *)namespace_name);
91 strcpy(tmp_name+namespace_name.len(),"__");
92 auto target = tmp_name+namespace_name.len()+2;
93 for( auto i : input) {
94 strcpy(target,i.name_);
95 insert( tmp_name, i.token_, i.type_);
96 }
97 if( also_load_global )
98 load(input);
99 }
100
101 virtual ~SymbolTableImpl() {}
102};

158Symbol * Lexer::unqualified_to_sym( int token_code, jString & token ) {
159 // What do we have here?
160 // All caps -> awk::
161 // AWK keyword -> awk::
162 // namespace == awk:: -> no prefix
163 // else namespace__token
164 if( ( namespace_prefix_.len()==0) || ( token.isalpha() && token.isupper() ) )
165 return symbol_table_->get(token, token_code, VARIABLE );
166
167 auto sym=symbol_table_->find(token);
168 if( sym ) {
169 // "Magic" types
170 switch( sym->type_ ) {
171 case STATEMENT:
172 case KEYWORD:
173 case OPERATOR:
174 return sym;
175 }
176 }
177 token=namespace_prefix_+token;
178 return symbol_table_->get(token, token_code, VARIABLE );
179}

181void Lexer::parseunqualified(int token_code, SymbolType type) { // identifier
182 if( expect_namespacename_)
183 start_namespace(token_code, type);
184 else {
185 size_t stringlen=buf_-tok_;
186 token_ = jString(tok_,buf_);
187 auto sym=unqualified_to_sym(token_code, token_);
188 token_ = sym->name_;
189 auto ast = new awkccc::ast_node(Expression, sym );
190 parser_->parse( sym->token_, ast, & ast_out );
191 allow_regex_ = false;
192 }
193}

195void Lexer::parsechr(char char_code) {
196 int token_code = parser_->char_to_token( char_code );
197 if( token_code < 1 ) {
198 // construct error message. Once c++17's std::string_view
199 // is universally available, switch to that.
200 auto buffer=new char[2];
201 buffer[0] = *tok_;
202 buffer[1]='\0';
203 std::cout << "* * * ERROR * * *" << ": \"" << buffer << "\"\n";
204 delete[] buffer;
205 } else {
206 parse( token_code, OPERATOR );
207 }
208}

210// Handle /.../ or /...(eol)
211// Posix requires a token starting / should be recognised as
212// division wherever a division is syntactically valid otherwise as a regex.
213// It was a nightmare trying to get a YACC style parser to inform the
214// lexer how to interpret a /, and despite it being really ugly to build
215// a subset of grammar rules into the lexer, at least it didn't make my
216// brain explode.
217// Basically: We recognise / as division only after a variable, a closing
218// bracket or parenthesis or a post increment or decrement.
219// allow_regex_ begins as true as program might begin: /[a-e]+/ {do_something;}
220// but can't begin with a division
221void Lexer::maybe_parse_regex(int token_code) {
222 if( allow_regex_)
223 parse(token_code,REGEX);
224 else if( tok_[1] == '=' ) { // could be /=
225 buf_ = tok_+2;
226 parse( PARSER_DIV_ASSIGN, OPERATOR);
227 } else {
228 static int divide = PARSER_char_to_token( '/' );
229 buf_ = tok_+1;
230 token_code=*tok_;
231 parse( divide, OPERATOR);
232 }
233}
