120 Lexer * create_lexer( const char * code ) {
121 SymbolTable & the_symbol_table = SymbolTable::instance();
122 PARSER_Parser * pParser=get_parser();
123 ParserMock * parser = static_cast<ParserMock *>(pParser);
124 parser->trace_.clear();
125 if( buf_ )
126 delete_buffer();
127 if( code ) {
128 buf_ = new char[std::strlen(code)+1];
129 std::strcpy( buf_, code );
130 } else {
131 buf_ = new char[2];
132 buf_[0] = buf_[1] = '\0';
133 }
134 Lexer * lexer = new Lexer( buf_ ,nullptr,nullptr,nullptr,nullptr,0,&the_symbol_table,parser);

136 return lexer;
137 }
