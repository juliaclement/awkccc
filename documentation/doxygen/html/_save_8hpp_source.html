27 * Purpose: Save<T> is a template class that stores a copy of a
28 * T variable's data and restores it when the Save<T>
29 * is destroyed.
30 * T must be copy constructable & assignable.

32 * Part of the JCLib project https://github.com/juliaclement/jclib
33 *
34 * Created on 7 November 2022
35 */
36#ifndef Save_HPP
37#define Save_HPP
38
39namespace jclib {
40 namespace Private {
41 template<typename T> class Saver {
42 private:
43 T & saved_object_;
44 T saved_data_;
45 /*** Occasionally we need to disable the restore */
46 bool enabled_ = true;
47 // forbid copying
48 Saver( const Saver & old ) = delete;
49 Saver & operator =( const Saver & old ) = delete;
50 public:
51 Saver(T& savee )
52 : saved_object_( savee )
53 , saved_data_( savee )
54 {}
55 ~Saver( ) {
56 if( enabled_ )
57 saved_object_ = saved_data_;
58 }
59 /*** restore data ahead of destruction.
60 * allows multiple restores of the
61 * original data, effectively creating a checkpoint.
62 */
63 Saver & restore( ) {
64 saved_object_ = saved_data_;
65 return * this;
66 }
67 /*** Turn off the automatic restore
68 * Typically used after a manual call to restore()
69 */
70 Saver & disable() {
71 enabled_ = false;
72 return * this;
73 }
74 /*** Turn on the automatic restore
75 * Only required following a disable()
76 */
77 Saver & enable() {
78 enabled_ = true;
79 return * this;
80 }
81
82 };
83 };
84 /*** Create an object that will capture the current value of
85 * savee, then restore this value when the created
86 * object is destroyed.
87 */
88 template<typename T> Private::Saver<T> Save( T& savee) {
89 return Private::Saver<T>(savee);
90 }
91};
