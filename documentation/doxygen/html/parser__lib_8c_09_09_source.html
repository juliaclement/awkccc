52 out_ << ctl_.padding_ << ctl_.extra_ << node->name_;

54 out_ << " (Lexer";

59 } else

215 // Normalise the sibling arrays by promoting each node's sibling's to be its
216 // parent's children.
217 // Requires walking the tree
218 void ast_node::clean_tree(ast_node * parent) {
219 if(sibling_nodes_.size() > 0 ) {
220 if( parent != nullptr && parent->extra_children_allowed_ ) {
221 // Move my siblings to parent's children
222 for( auto child : sibling_nodes_ ) {
223 parent->child_nodes_.push_back(child);
224 child->clean_tree( parent );
225 }
226 sibling_nodes_.clear();
227 }
228 else{
229 // Notify my siblings
230 for( auto child : sibling_nodes_ ) {
231 child->clean_tree( parent );
232 }
233 }
234 }
235 if( child_nodes_.size() > 0 ) {
236 if( extra_children_allowed_ ) {
237 // Add children's siblings.
238 // A bit complex as we may be inserting in the middle of the child array
239 // We solve this by copying to a temporary
240 std::vector<jclib::CountedPointer<ast_node> > temp_children(child_nodes_);
241 child_nodes_.clear();
242 for( auto child : temp_children ) {
243 child_nodes_.push_back(child);
244 child->clean_tree( this );
245 }
246 } else {
247 // Still process the children's children
248 for( auto child : child_nodes_) {
249 child->clean_tree( this );
250 }
251 }
252 // We may have dummy child notes that are empty
253 bool empty_children_exist = false;
254 for( auto child : child_nodes_) {
255 if (child->dummy_ && child->child_nodes_.empty() && child->sibling_nodes_.empty()) {
256 empty_children_exist = true;
257 break;
258 }
259 }
260 if( empty_children_exist ) {
261 std::vector<jclib::CountedPointer<ast_node> > temp_children(child_nodes_);
262 child_nodes_.clear();
263 for( auto child : temp_children) {
264 if( (!child->dummy_) || (!child->child_nodes_.empty()) || (! child->sibling_nodes_.empty()))
265 child_nodes_.push_back(child);
266 }
267 }
268 }
269 }
