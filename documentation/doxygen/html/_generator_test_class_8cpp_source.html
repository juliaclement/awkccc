104 Lexer * create_lexer( const char * code ) {
105 SymbolTable & the_symbol_table = SymbolTable::instance();
106 PARSER_Parser * pParser=get_parser();
107 PARSER_Parser * parser = static_cast<PARSER_Parser *>(pParser);
108 if( buf_ )
109 delete_buffer();
110 if( code ) {
111 buf_ = new char[std::strlen(code)+1];
112 std::strcpy( buf_, code );
113 } else {
114 buf_ = new char[2];
115 buf_[0] = buf_[1] = '\0';
116 }
117 Lexer * lexer = new Lexer( buf_ ,nullptr,nullptr,nullptr,nullptr,0,&the_symbol_table,parser);
118 lexer->initialise_symbol_table();
119 return lexer;
120 }
121 ast_node_ptr lex( const char * code ) {
122 Lexer * lexer = create_lexer( code );
123 lexer->lex();
124 if( Lexer::ast_out.isset()) {
125 Lexer::ast_out->clean_tree(nullptr);
126 }
127 ast_node_ptr node = Lexer::ast_out;
128 delete lexer;
129 delete_buffer();
130 return node;
131 }
