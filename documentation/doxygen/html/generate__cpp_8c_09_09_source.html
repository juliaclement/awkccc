243 void write_to_cpp( std::istream & template_text,
244 std::ostream & program_file,
245 std::vector< postream > & code ){
246 std::string markers[] = {"%includes;","%variables;","%procedures;","%begin;","%begin_file;","%body;","%end_file;","%end;"};
247 std::string line;
248 while( getline( template_text, line ) ){
249 if( line[0] == '%' ) {
250 bool found = false;
251 for( int i = 0; i <= template_END; ++i ) {
252 auto kw = markers[i];
253 if( line.compare(0,kw.length(),kw)) {
254 found = true;
255 std::string out = code[i]->str();
256 program_file << out;
257 break;
258 }
259 }
260 if( ! found )
261 program_file << line << "\n";
262 } else {
263 program_file << line << "\n";
264 }
265 }
266 }

285 void generate_cpp( std::istream & template_text, std::ostream & program_file, awkccc::ast_node * node ){
286 std::vector<postream> code;
287 while( code.size() < template_NR_SEGMENTS ) {
288 auto x = new std::ostringstream;
289 code.push_back(x);

291 generate_cpp(template_text,program_file, node);
292 for( auto x : code){
