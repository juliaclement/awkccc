173 void insert(CountedPointer<arg_base> input) {
174 all_options_.push_back( input );
175 if( input->short_name_.len()==1) {
176 if( short_names_.find(input->short_name_) != short_names_.end() ) {
177 error_log_ << "Internal error: \"" << input->short_name_ << "\" duplicated\n";
178 errors_++;
179 }
180 else{
181 short_names_[input->short_name_] = input;
182 input->owner_ = this;
183 }
184 } else if( input->short_name_.len() != 0 ) {
185 error_log_ << "Internal error: \"" << input->short_name_ << "\" longer than 1 character\n";
186 errors_++;
187 }
188 if( input->long_name_.len()>0) {
189 if( long_names_.find(input->long_name_) != long_names_.end() ) {
190 error_log_ << "Internal error: \"" << input->long_name_ << "\" duplicated\n";
191 errors_++;
192 }
193 else{
194 long_names_[input->long_name_] = input;
195 input->owner_ = this;
196 }
197 }
198 }

206 bool process_args( int argc, const char ** argv, int start_at = 1) {
207 int argnum = start_at;
208 for( ; argnum < argc && argv[ argnum ][0] == '-'; ++argnum ) {
209 const char * argstr=argv[argnum ];
210 if( argstr[1] == '-') { // long arg
211 if( argstr[2] == '\0') { // end of kw args
212 ++argnum;
213 break;
214 }
215 std::vector<jString> argarray = jString(argstr+2).split("=",1);
216 jString kw=argarray[0];
217 jString val=argarray[1];
218 if( ! val.isvalid())
219 val = jString::get_empty();
220 auto found = long_names_.find(kw);
221 if( found == long_names_.end()) {
222 error_log_<< "Unknown argument " << argstr << "\n";
223 errors_++;
224 }else{
225 arg_base & arg = * found->second;
226 if( arg.is_set_ && ! arg.multiples_ok_ ) {
227 error_log_<< "Multiple use of " << argstr << "\n";
228 errors_++;
229 continue;
230 }
231 if( arg.takes_parameter_) {
232 int val_len = val.len();
233 int nextargnum=argnum + 1;
234 if( (val_len==0) && (nextargnum < argc) && (argv[nextargnum][0] != '-')) {
235 val = argv[++argnum];
236 }
237 arg.is_set_ = arg.set(val);
238 } else {
239 if( val.len() == 0 )
240 arg.is_set_ = arg.set();
241 else {
242 error_log_ << arg.long_name_ << " does not accept options\n";
243 errors_++;
244 }
245 }
246 }
247 } else {
248 // args can be run together, so we have to decode the arg to know
249 // e.g. -abcdef might be -a(=bcdef) or -a -b -c(=def) or ...
250 const char * end = argstr + strlen(argstr);
251 // +1 to skip the -
252 for( const char *chars = argstr+1; chars < end; ++ chars ) {
253 auto kw=jString(chars,1);
254 auto found = short_names_.find(kw);
255 if( found == short_names_.end()) {
256 error_log_<< "Unknown argument " << argstr << "\n";
257 errors_++;
258 }else{
259 arg_base & arg = * found->second;
260 if( arg.is_set_ && ! arg.multiples_ok_ ) {
261 error_log_<< "Multiple use of " << argstr << "\n";
262 errors_++;
263 continue;
264 }
265 if( arg.takes_parameter_) {
266 jString val(chars+1);
267 if( chars[1] == '\0' && (argnum + 1) < argc && argv[argnum+1][0] != '-') {
268 val = argv[++argnum];
269 }
270 arg.is_set_ = arg.set(val);
271 break;
272 } else
273 arg.is_set_ = arg.set();
274 }
275 }
276 }
277 }
278 // Now process the remaining (positional) arguments
279 for( ; argnum < argc; ++ argnum) {
280 this->positional_args_.push_back(argv[argnum]);
281 }
282 // Finally, scan arguments to make sure required ones are set
283
284 if( ! (show_version_ || show_help_ ) )
285 {
286 for( auto k : all_options_) {
287 if( k->required_ && ! k->is_set_) {
288 errors_++;
289 error_log_ << "Required option \"";
290 if( k->short_name_.len() > 0 && k->long_name_.len() > 0 )
291 error_log_ << "--" << k->long_name_ << "\" (\"-" << k->short_name_ << "\")";
292 else if (k->short_name_.len() > 0)
293 error_log_ << "-" << k->short_name_ << "\"";
294 else
295 error_log_ << "--" << k->long_name_ << "\"";
296 if( k->desc_.len() > 0 )
297 error_log_ << " (" << k->desc_ << ")";
298 error_log_ << " missing\n";
299 errors_++;
300 }
301 }
302 }
303 // If version has been requested & we are set-up to provide it.
304 if( show_version_ ) {
305 error_log_ << version_string_ << "\n";;
306 }
307 // If help has been requested or a parameter error is detected & we are set-up to provide it.
308 if( show_help_ || ( errors_ != 0 && help_prefix_.len()>0 ) ) {
309 error_log_ << help_prefix_ << "\n";
310 // find longest long option name
311 size_t longest = 1;
312 for( auto k : all_options_) {
313 longest=std::max<size_t>( longest, k->long_name_.len() );
314 }
315 // now print the lines
316 longest ++;
317 for( auto k : all_options_) {
318 if( k->short_name_.len() > 0 )
319 error_log_ << " -" << k->short_name_ << " ";
320 else
321 error_log_<< " ";
322 error_log_ << (k->long_name_.len()==0 ? " " : "--")
323 << k->long_name_.pad(longest)
324 << k->desc_ << "\n";
325 }
326 error_log_ << help_suffix_ << "\n";
327 }
328 return errors_ == 0 && !(show_version_ || show_help_);
329 }
