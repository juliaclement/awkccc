42template< typename T> class CountedPointer {
43private:
44 T * ptr_;
45 inline void postAssign() {
46 if( ptr_ != nullptr ) {
47 ptr_->CountedPointerAttach();
48 }
49 }
50public:
55 bool isnull() const { return ptr_ == nullptr; }
60 bool isset() const { return ptr_ != nullptr; }
61
65 T * get() const { return const_cast< T *>(ptr_); }
69 T & operator *() const { return const_cast< T &>(*ptr_); }
70 T * operator ->() const { return const_cast< T *>(ptr_); }
71 operator T *() const { return const_cast< T *>(ptr_); }
77 inline void set( T * const ptr) {
78 // Note: We Attach before Detach so A=A (with attach count==1)
79 // doesn't cause the attach count to temporarily zero and have
80 // the target delete itself;
81 T * tmp_ptr=ptr_;
82 ptr_ = const_cast< T*>(ptr);
83 postAssign();
84 if( tmp_ptr != nullptr ) {
85 tmp_ptr->CountedPointerDetach();
86 }
87 }
93 inline void set( const CountedPointer<T> & old ) {
94 set( old.ptr_);
95 }
101 class CountedPointer<T> & operator =( T * const ptr ) {
102 set( ptr );
103 return *this;
104 }
110 class CountedPointer<T> & operator =( const CountedPointer<T> & ptr ) {
111 set( ptr.ptr_ );
112 return *this;
113 }
114 CountedPointer( ) : ptr_( nullptr) { postAssign(); }
115 CountedPointer( T * ptr ) : ptr_( ptr) { postAssign(); }
116 CountedPointer( const class CountedPointer<T>& old ) : ptr_( old.ptr_) { postAssign(); }
117 ~CountedPointer() { if( ptr_ != nullptr ) ptr_->CountedPointerDetach(); ptr_=nullptr; }
118};
